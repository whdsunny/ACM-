/*高精度乘法*/
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=1010;
int sum[maxn];
char num1[maxn],num2[maxn];
int multiplication(char num1[],char num2[],int sum[]){
	int len,i,j;
	int a[maxn],b[maxn],c[maxn];
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	memset(c,0,sizeof(c));
	int len1=strlen(num1);
	int len2=strlen(num2);
	for(i=len1-1,j=0;i>=0;i--,j++){
		a[j]=num1[i]-'0';
	}
	for(i=len2-1,j=0;i>=0;i--,j++){
		b[j]=num2[i]-'0';
	}
	for(i=0;i<len2;i++){
		for(j=0;j<len1;j++){
			c[i+j]+=b[i]*a[j];
		}
	}
	for(i=0;i<maxn*2;i++){
		if(c[i]>=10){
			c[i+1]+=c[i]/10;
			c[i]%=10;
		}
	} 
	for(i=maxn*2;c[i]==0&&i>=0;i--)
		;
	len=i+1;
	for(;i>=0;i--){
		sum[i]=c[i];
	}
	return len;
}
int main(){
	int len;
	int sum[maxn*2+10];
	memset(sum,0,sizeof(sum));
	cin>>num1>>num2;
	len=multiplication(num1,num2,sum);
	for(int i=len-1;i>=0;i--){
		printf("%d",sum[i]);
	}
	printf("\n");
	return 0;
}
